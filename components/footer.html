Basic Footer

<footer class="main-footer">
  <div class="content">
    <div class="brand">
      <div class="logo">Company</div>
      <p class="tagline">Your tagline here</p>
    </div>

    <div class="links">
      <div class="column">
        <h4>Product</h4>
        <ul>
          <li><a href="#">Features</a></li>
          <li><a href="#">Pricing</a></li>
        </ul>
      </div>
    </div>
  </div>

  <div class="bottom">
    <p>© 2025 Company. All rights reserved.</p>
  </div>
</footer>
