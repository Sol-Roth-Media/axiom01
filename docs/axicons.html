Axicons

3,969 SVG icons across 5 variants. Click any icon to copy its name.

Getting Started

Load only what you need. Start with the base set and add variants as required.

HTML Markup

<!-- Basic icon -->
<span class="axicon render" data-name="Check"></span>

<!-- Color variant -->
<span class="axicon render" data-name="Check-Color"></span>

<!-- Premium variant -->
<span class="axicon render" data-name="Check-Premium"></span>

Script Loading

<!-- Base (89KB, 450 icons) - always load -->
<script src="js/axicons-base.js"></script>

<!-- Optional variants -->
<script src="js/axicons-thin-variants.js"></script>
<script src="js/axicons-color-variants.js"></script>
<script src="js/axicons-premium-variants.js"></script>

<!-- Renderer (always last) -->
<script src="js/render-icons.js"></script>

Styling

/* Icons inherit currentColor */
.axicon svg {
  width: 1em;
  height: 1em;
}

/* Color via parent */
button {
  color: var(--a-color-primary);
}

/* Explicit size */
.icon-lg .axicon { font-size: 2rem; }
