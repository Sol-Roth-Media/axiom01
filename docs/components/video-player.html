  <source src="video.mp4" type="video/mp4">
  Your browser does not support the video tag.
</video>

Video with Poster Image

Video with Captions

<video controls width="640" height="360">
  <source src="video.mp4" type="video/mp4">
  <track kind="subtitles" src="en.vtt" srclang="en" label="English">
  <track kind="captions" src="en-cc.vtt" srclang="en" label="English (Captions)">
  Your browser does not support the video tag.
</video>

Video Attributes

<video 
  controls            <!-- Show player controls -->
  autoplay            <!-- Start playing automatically -->
  loop                <!-- Loop when finished -->
  muted               <!-- Mute by default -->
  poster="img.jpg"    <!-- Thumbnail image -->
  width="640" 
  height="360"
  preload="metadata"  <!-- none, metadata, auto -->
>
  <source src="video.mp4" type="video/mp4">
</video>

JavaScript API

Basic Controls

const video = document.querySelector('video');

// Play/pause
video.play();
video.pause();

// Get current time and duration
console.log(video.currentTime);  // seconds
console.log(video.duration);     // seconds

// Set playback position
video.currentTime = 30;

// Volume control (0 to 1)
video.volume = 0.5;
video.muted = true;

// Playback rate
video.playbackRate = 1.5; // 1.5x speed

Event Handling

const video = document.querySelector('video');

// Listen for events
video.addEventListener('play', () => console.log('Playing'));
video.addEventListener('pause', () => console.log('Paused'));
video.addEventListener('ended', () => console.log('Finished'));
video.addEventListener('timeupdate', () => {
  const percent = (video.currentTime / video.duration) * 100;
  console.log('Progress:', percent + '%');
});

video.addEventListener('loadedmetadata', () => {
  console.log('Duration:', video.duration);
});

video.addEventListener('error', (e) => {
  console.log('Error:', e.target.error.code);
});

Accessibility

Best Practices

  • Always provide captions for deaf/hard of hearing users
  • Include audio descriptions for blind users
  • Use WebVTT format for subtitles
  • Test keyboard navigation (space, arrows, escape)
  • Provide fallback text for unsupported browsers
  • Use semantic markup around video
  • Test with screen readers

AXIOM01 Philosophy

Why Use Native <video>

The native <video> element embodies AXIOM01 principles:

  • Semantic: <video> tells the browser exactly what it is
  • Accessible: Built-in controls work with keyboard and screen readers
  • Minimal Code: No custom JavaScript needed for basic playback
  • Progressive Enhancement: Works without JavaScript, enhanced with it
  • Browser Native: Uses OS-level video codecs and controls
  • Responsive: Scales with container automatically

Custom video players require hundreds of lines of code and often break accessibility. Use <video> first.
