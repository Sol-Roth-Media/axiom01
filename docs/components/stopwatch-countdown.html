Production data binding pattern

Bind timer defaults from user/task records and persist edits through semantic form controls instead of utility wrappers.

const timer = document.querySelector('[data-component="timer"]');
const mode = timer.querySelector('select');
const minutes = timer.querySelector('[name="minutes"]');
const seconds = timer.querySelector('[name="seconds"]');

const schedule = await fetch('/api/sessions/active-timer').then((res) => res.json());
mode.value = schedule.mode;
minutes.value = schedule.minutes;
seconds.value = schedule.seconds;

timer.addEventListener('change', () => {
  fetch('/api/sessions/active-timer', {
    method: 'PATCH',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ mode: mode.value, minutes: minutes.value, seconds: seconds.value })
  });
});
