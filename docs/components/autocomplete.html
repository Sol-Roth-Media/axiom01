HTML Structure

<div class="field" data-type="autocomplete">
				<label for="search">Search</label>

				<input
				type="text"
				id="search"
				placeholder="Start typing..."
				aria-autocomplete="list"
				aria-controls="dropdown"
				>

				<ul id="dropdown" role="listbox">
				<li role="option">Option 1</li>
				<li role="option">Option 2</li>
				<li role="option">Option 3</li>
				</ul>
				</div>

Data Attributes

  • `data-type="autocomplete"` - Container for autocomplete field
  • `data-open="true"` - Controls listbox visibility
  • `data-active="true"` - Marks highlighted option

ARIA Attributes

  • `aria-autocomplete="list"` - Input provides autocomplete list
  • `aria-controls` - Links input to listbox
  • `role="listbox"` - Container for autocomplete options
  • `role="option"` - Individual selectable items

Key Features

  • Real-time Filtering - Options filter as user types
  • Keyboard Navigation - Arrow keys, Enter, Escape support
  • Mouse Support - Click to select options
  • No Results Feedback - Message when nothing matches
  • Accessible - Full screen reader support with ARIA
  • Semantic HTML - Native `
      `, `
    • ` elements

JavaScript Implementation

function setupAutocomplete(inputId, listboxId) {
				const input = document.getElementById(inputId);
				const listbox = document.getElementById(listboxId);
				const options = listbox.querySelectorAll('[role="option"]');
				const allText = Array.from(options).map(o => o.textContent);
				let activeIndex = -1;

				input.addEventListener('input', (e) => {
				const query = e.target.value.toLowerCase();
				const matches = query
				? allText.filter(t => t.toLowerCase().includes(query))
				: [];

				listbox.innerHTML = '';

				if (query && matches.length) {
				matches.forEach(text => {
				const li = document.createElement('li');
				li.setAttribute('role', 'option');
				li.textContent = text;
				li.addEventListener('click', () => {
				input.value = text;
				listbox.setAttribute('data-open', 'false');
				});
				listbox.appendChild(li);
				});
				listbox.setAttribute('data-open', 'true');
				} else if (query) {
				const none = document.createElement('li');
				none.className = 'no-results';
				none.textContent = 'No results';
				listbox.appendChild(none);
				listbox.setAttribute('data-open', 'true');
				} else {
				listbox.setAttribute('data-open', 'false');
				}
				activeIndex = -1;
				});

				input.addEventListener('keydown', (e) => {
				const visible = listbox.querySelectorAll('[role="option"]');

				if (e.key === 'ArrowDown') {
				e.preventDefault();
				activeIndex = Math.min(activeIndex + 1, visible.length - 1);
				updateActive(visible);
				} else if (e.key === 'ArrowUp') {
				e.preventDefault();
				activeIndex = Math.max(activeIndex - 1, -1);
				updateActive(visible);
				} else if (e.key === 'Enter' && activeIndex >= 0) {
				e.preventDefault();
				input.value = visible[activeIndex].textContent;
				listbox.setAttribute('data-open', 'false');
				} else if (e.key === 'Escape') {
				listbox.setAttribute('data-open', 'false');
				}
				});

				function updateActive(visible) {
				visible.forEach((opt, i) => {
				opt.setAttribute('data-active', i === activeIndex ? 'true' : 'false');
				});
				}
				}

				setupAutocomplete('country', 'country-options');
				setupAutocomplete('tech', 'tech-options');

Best Practices

  • Debounce - For large datasets, debounce input filtering
  • Case Insensitive - Match regardless of case
  • Partial Match - Match anywhere in string, not just start
  • Min Characters - Optionally wait for 2-3 chars before filtering
  • Clear Feedback - Show "no results" when nothing matches
  • Max Results - Limit displayed options (e.g., 10) for performance
  • Keyboard Only - Full navigation without mouse
